                              RECURSE

         TARGET_NAME_VARIABLE sRrcxxLib
                              SHARED
                              ${zSrc}

          INCLUDE_DIRECTORIES PUBLIC  "../"
                              PRIVATE "${CMAKE_CURRENT_BINARY_DIR}/../")
