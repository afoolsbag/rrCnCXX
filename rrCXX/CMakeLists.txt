  zGenerated
  DIRECTORY  "${CMAKE_CURRENT_BINARY_DIR}/"
  GROUP_ROOT "generated/")

  NAME "test"
  ${zSource} ${zGenerated}
  CXX17
  INCLUDE_DIRECTORIES PRIVATE "./"
                              "${CMAKE_CURRENT_BINARY_DIR}/"
                              ${CCGSL_INCLUDE_DIRS})
